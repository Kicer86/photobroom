44template<typename MapT,
45 typename RemovedT,
46 typename ChangedT,
47 typename AddedT>
48void compare(const MapT& lhs, const MapT& rhs,
49 RemovedT rem_inserter,
50 ChangedT chg_inserter,
51 AddedT add_inserter)
52{
53 typedef typename MapT::value_type Pair;
54 typedef std::tuple<typename MapT::key_type,
55 typename MapT::mapped_type,
56 typename MapT::mapped_type> Changed;
57
58 // find *keys* which exist only in lhs
59 std::set_difference(lhs.cbegin(), lhs.cend(),
60 rhs.cbegin(), rhs.cend(),
61 rem_inserter, [](const Pair& l, const Pair& r)
62 {
63 return l.first < r.first;
64 });
65
66 // find *keys* which exist only in rhs
67 std::set_difference(rhs.cbegin(), rhs.cend(),
68 lhs.cbegin(), lhs.cend(),
69 add_inserter, [](const Pair& l, const Pair& r)
70 {
71 return l.first < r.first;
72 });
73
74 // find *keys* which exist in both collections
75 std::vector<Pair> intersection;
76 std::set_intersection(lhs.cbegin(), lhs.cend(),
77 rhs.cbegin(), rhs.cend(),
78 std::back_inserter(intersection), [](const Pair& l, const Pair& r)
79 {
80 return l.first < r.first;
81 });
82
83 for(const Pair& lhs_data: intersection)
84 {
85 auto it = rhs.find(lhs_data.first);
86 assert(it != rhs.cend());
87
88 if (lhs_data.second != it->second)
89 {
90 Changed changed(lhs_data.first, lhs_data.second, it->second);
91
92 chg_inserter = changed;
93 chg_inserter++;
94 }
95 }
96
97}

101{

103
104 std::vector<T> result(size);
105

107 {
108 const T l = i < lhs.size()? lhs[i]: T();

148template<typename ForwardIt, typename BinaryPredicate>
149ForwardIt remove_unique(ForwardIt first, ForwardIt last, BinaryPredicate p)
150{
151 if (std::distance(first, last) < 2)
152 return first;
153
154 ForwardIt result = first;
155
156 for(auto it = first; it != last; ++it)
157 {
158 bool nonUnique = false;
159
160 if (it != first)
161 {
162 auto prev = std::prev(it);
163
164 nonUnique = p(*prev, *it);
165 }
166
167 if (nonUnique == false)
168 {
169 auto next = std::next(it);
170
171 if (next != last)
172 nonUnique = p(*next, *it);
173 }
174
175 if (nonUnique)
176 {
177 if (result != it)
178 *result = std::move(*it);
179
180 ++result;
181 }
182 }
183
184 return result;
185}
