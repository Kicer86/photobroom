48 template<typename T, typename L, typename ...Args>
49 void execute(ILogger* logger,
50 const QString& executable,
51 T&& outputDataCallback,
52 L&& launcher,
53 const Args... args)
54 {
55 QStringList arguments;
56 append(arguments, args...);
57
58 QProcess pr;
59 pr.setProcessChannelMode(QProcess::MergedChannels);
60
61 QObject::connect(&pr, &QIODevice::readyRead, [&outputDataCallback, &pr]()
62 {
63 outputDataCallback(pr);
64 });
65
66 pr.setProgram(executable);
67 pr.setArguments(arguments);
68
69 const QString info_message = QString("Executing %1 %2").arg(executable).arg(arguments.join(" "));
70
71 logger->info(info_message);
72
73 launcher(pr);
74 }
