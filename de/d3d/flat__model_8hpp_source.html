89 template<typename T>
90 decltype(m_photos)::iterator insertPhotos(decltype(m_photos)::iterator position_it, T first, T last)
91 {
92 if (first == last)
93 return position_it;
94
95 const auto items = static_cast<int>(std::distance(first, last));
96 const auto position = static_cast<int>(std::distance(m_photos.begin(), position_it));
97
98 beginInsertRows({}, position, position + items - 1);
99 auto r = m_photos.insert(position_it, first, last);
100 endInsertRows();
101
102 return r;
103 }
104
105 decltype(m_photos)::iterator erasePhotos(decltype(m_photos)::iterator first, decltype(m_photos)::iterator last)
106 {
107 if (first == last)
108 return first;
109
110 const auto items = static_cast<int>(std::distance(first, last));
111 const auto position = static_cast<int>(std::distance(m_photos.begin(), first));
112
113 beginRemoveRows({}, position, position + items - 1);
114
115 std::for_each(first, last, [this](auto& id){
116 m_properties.erase(id);
117 });
118
119 auto r = m_photos.erase(first, last);
120 endRemoveRows();
121
122 return r;
123 }
