21// safe_callback is generated by safe_callback_ctrl.
22// it will be a valid functor as long as safe_callback_ctrl exists.
23// When safe_callback_ctrl is destroyed, all its safe_callbacks will become
24// empty. operator() will do nothing.
25// Usefull when calling object's method from another thread
26// as we are sure target object exists (or nothing happens)
27template<typename... Args>
28class safe_callback final
29{
30 public:
31 safe_callback(const std::shared_ptr<safe_callback_data>& data, const std::function<void(Args...)>& callback): m_data(data), m_callback(callback) {}
32 safe_callback(const safe_callback<Args...> &) = default;
33
34 safe_callback& operator=(const safe_callback<Args...> &) = default;
35
36 virtual ~safe_callback() = default;
37
38 void operator()(Args... args) const
39 {
40 std::lock_guard<std::mutex> lock(m_data->mutex);
41
42 if (is_valid())
43 m_callback(std::forward<Args>(args)...);
44 }
45
46 bool is_valid() const
47 {
48 return m_data->callbackAlive == true && m_callback;
49 }
50
51 private:
52 std::shared_ptr<safe_callback_data> m_data;

138// call_from_object_thread uses Qt mechanisms to invoke function in another thread
139// (thread of 'object' object)
140template<typename F, typename ObjT, typename... Args>
141requires std::is_base_of_v<QObject, ObjT>
142void call_from_object_thread(QPointer<ObjT> object, const F& function, Args&&... args)
143{
144 if (object.data() != nullptr)
145 {
146 if constexpr (std::is_member_function_pointer_v<F>)
147 QMetaObject::invokeMethod(object.data(), [object, function, args...]()
148 {
149 ((*object).*function)(args...);
150 });
151 else
152 QMetaObject::invokeMethod(object.data(), [function, args...]()
153 {
154 function(args...);
155 });
156 }
157}

197// construct a functor which invoked will invoke a method
198// (slot) of given object. Will do nothing when given object is destroyed.
199// Similar to safe_callback_ctrl (but method will be invoked in target's thread)
200template<typename ObjT, typename R, typename ...Args>
201requires std::is_base_of_v<QObject, ObjT>
202auto queued_slot(ObjT* obj, R(ObjT::*method)(Args...))
