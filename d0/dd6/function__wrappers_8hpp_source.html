24// safe_callback is generated by safe_callback_ctrl.
25// it will be a valid functor as long as safe_callback_ctrl exists.
26// When safe_callback_ctrl is destroyed, all its safe_callbacks will become
27// empty. operator() will do nothing.
28// Usefull when calling object's method from another thread
29// as we are sure target object exists (or nothing happens)
30template<typename... Args>
31class safe_callback final
32{
33 public:
34 safe_callback(const std::shared_ptr<safe_callback_data>& data, std::function<void(Args...)>&& callback): m_data(data), m_callback(callback) {}
35 safe_callback(const safe_callback<Args...> &) = default;
36
37 safe_callback& operator=(const safe_callback<Args...> &) = default;
38
39 ~safe_callback()
40 {
41 std::lock_guard lock(m_data->mutex);
42 m_callback = std::function<void(Args...)>(); // release all possible resources held by m_callback
43 }
44
45 void operator()(Args... args) const
46 {
47 std::lock_guard lock(m_data->mutex);
48
49 if (is_valid())
50 m_callback(std::forward<Args>(args)...);
51 }
52
53 bool is_valid() const
54 {
55 return m_data->callbackAlive == true && m_callback;
56 }
57
58 private:
59 std::shared_ptr<safe_callback_data> m_data;
60 std::function<void(Args...)> m_callback;
61};

64// safe_callback_ctrl controls and creates safe_callbacks
65class safe_callback_ctrl final
66{
67 public:
68 safe_callback_ctrl(): m_data()
69 {
70 setup();
71 }

73 safe_callback_ctrl(const safe_callback_ctrl &) = delete;
74
75 ~safe_callback_ctrl()
76 {
77 // safe_callback_ctrl should be reset before its destruction.
78 //
79 // Object owning instance of this class should call invalidate()
80 // as a first action in its destructor.
81 // Otherwise owner class may be partialy destructed before
82 // safe_callback_ctrl is reseted which will allow
83 // callbacks to work on a broken object
84 auto data = m_data.load();
85 assert(data.get() == nullptr || data.use_count() == 2); // no data, or only one client - us (there are 2 copies of data - `m_data` and `data`)
86
87 reset();
88 }

90 template<typename... R, typename T>
91 [[nodiscard]] auto make_safe_callback(T&& callback) const
92 {
93 auto data = m_data.load();
94
95 std::lock_guard lock(data->mutex);
96 safe_callback<R...> callbackPtr(data, std::forward<T>(callback));
97
98 return callbackPtr;
99 }

101 safe_callback_ctrl& operator=(const safe_callback_ctrl &) = delete;
102
103 void invalidate()
104 {
105 reset(); // dissolve all connections
106 setup(); // create new one
107 }
108
109 private:
110 template<typename...>
111 friend class safe_callback;
112
113 std::atomic<std::shared_ptr<safe_callback_data>> m_data;
114
115 void setup()
116 {
117 m_data.store(std::make_shared<safe_callback_data>());
118 }
119
120 void reset()
121 {
122 // mark all safe callbacks as invalid
123 {
124 // make copy of shared_ptr - reset() may be called from any thread, so we cannot operate on the same instance of m_data
125 // as main thread. Operate on a copy instead which is thread safe
126 auto data = m_data.load();
127
128 // lock resource
129 std::lock_guard lock(data->mutex);
130
131 // mark resource as dead
132 data->callbackAlive = false;
133 }
134
135 // replace m_data with empty shared_ptr (as result, resetting m_data)
136 m_data.store(std::shared_ptr<safe_callback_data>());
137 }
138};

184// call_from_object_thread uses Qt mechanisms to invoke function in another thread
185// (thread of 'object' object)
186template<typename F, typename ObjT, typename... Args>
187requires std::is_base_of_v<QObject, ObjT>
188void call_from_object_thread(QPointer<ObjT> object, const F& function, Args&&... args)
189{
190 if (object.data() != nullptr)
191 {
192 if constexpr (std::is_member_function_pointer_v<F>)
193 QMetaObject::invokeMethod(object.data(), [object, function, args...]()
194 {
195 ((*object).*function)(args...);
196 });
197 else
198 QMetaObject::invokeMethod(object.data(), [function, args...]()
199 {
200 function(args...);
201 });
202 }
203}

295{
296 // Helper class containing safe_callback_ctrl related stuff.
297 // It's purpose is to live as long as task and callback are alive to make sure
298 // safe_callback_ctrl and std::stop_callback are alive as well (used by stop_token stop request).
299 class Context final
300 {
301 public:
302 explicit Context(std::stop_token stop_token)
303 : stop_callback(
304 stop_token,
305 [this]
306 {
307 ctrl.invalidate();
308 callback_invalidated = true;
309 })
310 {
311
312 }
313
314 ~Context()
315 {
316 // task should called callback (if valid)
317 assert(callback_called || callback_invalidated);
318
319 // ctrl needs to be invalidated before destruction
320 ctrl.invalidate();
321 }
322
323 safe_callback_ctrl ctrl;
324 std::stop_callback<std::function<void()>> stop_callback;
325 bool callback_called = false;
326 bool callback_invalidated = false;
327 };
328
329 const auto stop_token = stop_source.get_token();
330 auto context = std::make_shared<Context>(stop_token);
331
332 // inject context into callback so it will live with it and wrap it with safe_callback so
333 // it can be invalidated when stop (on stop_token) is requested.
334 auto safe_callback = context->ctrl.template make_safe_callback<R>(
335 [context, callback](const R& r)
336 {
337 callback(r);
338
339 // mark callback called
340 context->callback_called = true;
341 });
342
343 if (sameThread)
344 {
345 auto thisThreadCallback = make_cross_thread_function<R>(QThread::currentThread(), safe_callback);
346 task(stop_token, thisThreadCallback);
347 }
348 else
349 task(stop_token, safe_callback);
350};
