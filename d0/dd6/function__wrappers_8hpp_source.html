24// safe_callback is generated by safe_callback_ctrl.
25// it will be a valid functor as long as safe_callback_ctrl exists.
26// When safe_callback_ctrl is destroyed, all its safe_callbacks will become
27// empty. operator() will do nothing.
28// Usefull when calling object's method from another thread
29// as we are sure target object exists (or nothing happens)
30template<typename... Args>
31class safe_callback final
32{
33 public:
34 safe_callback(const std::shared_ptr<safe_callback_data>& data, std::function<void(Args...)>&& callback): m_data(data), m_callback(callback) {}
35 safe_callback(const safe_callback<Args...> &) = default;
36
37 safe_callback& operator=(const safe_callback<Args...> &) = default;
38
39 virtual ~safe_callback() = default;
40
41 void operator()(Args... args) const
42 {
43 std::lock_guard<std::mutex> lock(m_data->mutex);
44
45 if (is_valid())
46 m_callback(std::forward<Args>(args)...);
47 }
48
49 bool is_valid() const
50 {
51 return m_data->callbackAlive == true && m_callback;
52 }
53
54 private:
55 std::shared_ptr<safe_callback_data> m_data;
56
57 std::function<void(Args...)> m_callback;
58};

162// call_from_object_thread uses Qt mechanisms to invoke function in another thread
163// (thread of 'object' object)
164template<typename F, typename ObjT, typename... Args>
165requires std::is_base_of_v<QObject, ObjT>
166void call_from_object_thread(QPointer<ObjT> object, const F& function, Args&&... args)
167{
168 if (object.data() != nullptr)
169 {
170 if constexpr (std::is_member_function_pointer_v<F>)
171 QMetaObject::invokeMethod(object.data(), [object, function, args...]()
172 {
173 ((*object).*function)(args...);
174 });
175 else
176 QMetaObject::invokeMethod(object.data(), [function, args...]()
177 {
178 function(args...);
179 });
180 }
181}

275 // It's purpose is to live as long as task and callback are alive to make sure
276 // safe_callback_ctrl and std::stop_callback are alive as well (used by stop_token stop request).
277 class Context final

279 public:

281 : stop_callback(
282 stop_token,
283 [this]
284 {
285 ctrl.invalidate();
286 callback_invalidated = true;
287 })
288 {
289
290 }
291
292 ~Context()
293 {
294 // task should called callback (if valid)
295 assert(callback_called || callback_invalidated);
296
297 // ctrl needs to be invalidated before destruction
298 ctrl.invalidate();
299 }
300
301 safe_callback_ctrl ctrl;
302 std::stop_callback<std::function<void()>> stop_callback;
303 bool callback_called = false;
304 bool callback_invalidated = false;
305 };
306
307 const auto stop_token = stop_source.get_token();
308 auto context = std::make_shared<Context>(stop_token);
309
310 // inject context into callback so it will live with it and wrap it with safe_callback so
311 // it can be invalidated when stop (on stop_token) is requested.
312 auto safe_callback = context->ctrl.template make_safe_callback<R>(
313 [context, callback](const R& r)
314 {
315 callback(r);
316
317 // mark callback called
318 context->callback_called = true;
319 });
320
321 if (sameThread)
322 {
323 auto thisThreadCallback = make_cross_thread_function<R>(QThread::currentThread(), safe_callback);

325 }
326 else
327 task(stop_token, safe_callback);
