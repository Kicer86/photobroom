58// call_from_object_thread uses Qt mechanisms to invoke function in another thread
59// (thread of 'object' object)
60template<typename F, typename ObjT, typename... Args>
61requires std::is_base_of_v<QObject, ObjT>
62void call_from_object_thread(QPointer<ObjT> object, const F& function, Args&&... args)
63{
64 if (object.data() != nullptr)
65 {
66 if constexpr (std::is_member_function_pointer_v<F>)
67 QMetaObject::invokeMethod(object.data(), [object, function, args...]()
68 {
69 ((*object).*function)(args...);
70 });
71 else
72 QMetaObject::invokeMethod(object.data(), [function, args...]()
73 {
74 function(args...);
75 });
76 }
77}
