16// from https://en.cppreference.com/w/cpp/types/numeric_limits/epsilon
17template<class T>
18bool almost_equal(T x, T y, int ulp = 2)
19requires (std::numeric_limits<T>::is_integer == false)
20{
21 // the machine epsilon has to be scaled to the magnitude of the values used
22 // and multiplied by the desired precision in ULPs (units in the last place)
23 return std::fabs(x-y) <= std::numeric_limits<T>::epsilon() * std::fabs(x+y) * ulp
24 // unless the result is subnormal
25 || std::fabs(x-y) < std::numeric_limits<T>::min();
26}
