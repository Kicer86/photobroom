51// Helper function.
52// Run a task and wait for it to be finished.
53template<typename E, typename T>
54[[nodiscard]]
55auto evaluate(E& executor, const T& task)
56{
57 std::packaged_task p_task(task);
58
59 auto result_future = p_task.get_future();
60 ExecutorTraits<E, decltype(p_task)>::exec(executor, std::move(p_task));
61
62 result_future.wait();
63
64 return result_future.get();
65}

222 void waitForPendingTasks()
223 {
224 // wait for tasks being executed
225 std::unique_lock<std::recursive_mutex> lock(m_tasksMutex);
226 m_noWork.wait(lock, [this]
227 {
228 return m_executingTasks == 0;
229 });
230 }
231
232 protected:
233 typedef std::shared_ptr<void> Notifier;
234 virtual void passTaskToExecutor(Task&& task, const Notifier &) = 0;
235
236 private:
237 std::recursive_mutex m_tasksMutex;
238 std::deque<Task> m_waitingTasks;
239 std::condition_variable_any m_noWork;
240 int m_maxTasks = 0;
241 int m_executingTasks = 0;
242 const Mode m_mode = Mode::Fifo;
243
244 void task_finished_notification(void *)
245 {
246 task_finished();
247 }
248
249 void try_to_fire()
250 {
251 std::lock_guard<std::recursive_mutex> guard(m_tasksMutex);
252
253 // Do not put all waiting tasks to executor.
254 while (m_maxTasks > m_executingTasks && m_waitingTasks.empty() == false)
255 {
256 fire();
257 }
258 }
259
260 void fire()
261 {
262 std::lock_guard<std::recursive_mutex> guard(m_tasksMutex);
263 assert(m_waitingTasks.empty() == false);
264
265 auto task = m_mode == Mode::Fifo? take_front(m_waitingTasks): take_back(m_waitingTasks);
266
267 m_executingTasks++;
268 auto notifier = Notifier(nullptr, std::bind(&Queue::task_finished_notification, this, std::placeholders::_1));
269 passTaskToExecutor(std::move(task), notifier);
270 }
271
272 void task_finished()
273 {
274 // It is possible that in this function TasksQueue::fire() will be on stack
275 // of current thread (if addTask(std::move(task), notifier); executes task immediately)
276 // That's why we need recursive mutex here
277 std::lock_guard<std::recursive_mutex> guard(m_tasksMutex);
278
279 assert(m_executingTasks > 0);
280 m_executingTasks--;
281
282 if (m_executingTasks == 0)
283 m_noWork.notify_one();
284
285 try_to_fire();
286 }
287};
