37 public:
38 template<typename Fire>
39 explicit SignalPostponer(Fire fire, QObject* p):
40 QObject(p),
41 m_delay(std::chrono::milliseconds(250)),
42 m_patience(std::chrono::milliseconds(1000))
43 {
44 connect(&m_lazinessTimer, &QTimer::timeout, fire);
45 connect(&m_patienceTimer, &QTimer::timeout, fire);
46
47 // when any of timers timeouts, stop both of them
48 connect(&m_lazinessTimer, &QTimer::timeout, this, &SignalPostponer::stop);
49 connect(&m_patienceTimer, &QTimer::timeout, this, &SignalPostponer::stop);

51 m_lazinessTimer.setSingleShot(true);
52 m_patienceTimer.setSingleShot(true);
53 }
